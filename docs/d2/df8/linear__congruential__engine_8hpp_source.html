37  template <typename TIntType, TIntType a, TIntType c, TIntType m>
38  class linear_congruential_engine {
39  public:
40  using result_type = TIntType;
41  static constexpr result_type multiplier = a;
42  static constexpr result_type increment = c;
43  static constexpr result_type modulus = m;
44  static constexpr result_type default_seed = 1u;
45 
46  explicit linear_congruential_engine(result_type s = default_seed)
47  : m_iX(s % modulus) {}
48 
49  void seed(result_type s = default_seed) {
50  m_iX = s % modulus;
51  }
52  void discard(unsigned long long z) {
53  for (; z != 0ULL; --z)
54  (*this).operator ()();
55  }
56  result_type last() {
57  return m_iX;
58  }
59  result_type operator()() {
60  m_iX = ( multiplier * m_iX + increment) % modulus;
61  return m_iX;
62  }
63  static constexpr result_type min() { return c == 0u ? 1u: 0u; }
64  static constexpr result_type max() { return m - 1u; }
65  private:
66  result_type m_iX;
67  };
68  template <class UIntType, UIntType a, UIntType c, UIntType m>
69  bool operator == ( const linear_congruential_engine<UIntType,a,c,m>& lhs,
70  const linear_congruential_engine<UIntType,a,c,m>& rhs ) {
71  if(lhs.multiplier != rhs.multiplier) return false;
72  if(lhs.increment != rhs.increment) return false;
73  if(lhs.modulus != rhs.modulus) return false;
74  return true;
75  }

77  bool operator!= ( const linear_congruential_engine<UIntType,a,c,m>& lhs,
78  const linear_congruential_engine<UIntType,a,c,m>& rhs ) {
79  return !(lhs == rhs);
80  }
81 }
