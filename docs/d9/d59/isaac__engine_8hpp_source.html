37  template <class TIntType, size_t size, TIntType TGoldenRatio>
38  class isaac_engine {
39  public:
40  using result_type = TIntType;
41  static constexpr result_type golden_ratio = result_type(TGoldenRatio);

43  isaac_engine(result_type a=0, result_type b=0, result_type c =0) {
44  seed(a,b,c);
45  }
46  void seed(result_type a, result_type b, result_type c, result_type s =((void*)0)) {
47  result_type a,b,c,d,e,f,g,h;
48  a = b = c = d = e = f = g = h = golden_ratio;
49 
50  for(int i = 0; i < 256; i++) {
51  m_rsl[i] = s != ((void*)0) ? s[i] : 0;
52  }
53  m_a = a;
54  m_b = b;
55  m_c = c;
56 
57  shuffle(a,b,c,d,e,f,g,h);
58  shuffle(a,b,c,d,e,f,g,h);
59  shuffle(a,b,c,d,e,f,g,h);
60  shuffle(a,b,c,d,e,f,g,h);
61 
62 
63  for(result_type i=0; i < 256; i+=8) {
64  a+= (m_rsl)[i ]; b+=(m_rsl)[i+1]; c+=(m_rsl)[i+2]; d+=(m_rsl)[i+3];
65  e+= (m_rsl)[i+4]; f+=(m_rsl)[i+5]; g+=(m_rsl)[i+6]; h+=(m_rsl)[i+7];
66 
67  shuffle(a,b,c,d,e,f,g,h);
68 
69  (m_mem)[i ]=a; (m_mem)[i+1]=b; (m_mem)[i+2]=c; (m_mem)[i+3]=d;
70  (m_mem)[i+4]=e; (m_mem)[i+5]=f; (m_mem)[i+6]=g; (m_mem)[i+7]=h;
71 
72  }
73  for(result_type i=0; i < 256; i += 8) {
74  a+=(m_mem)[i ]; b+=(m_mem)[i+1]; c+=(m_mem)[i+2]; d+=(m_mem)[i+3];
75  e+=(m_mem)[i+4]; f+=(m_mem)[i+5]; g+=(m_mem)[i+6]; h+=(m_mem)[i+7];
76 
77  shuffle(a,b,c,d,e,f,g,h);
78 
79  (m_mem)[i ]=a; (m_mem)[i+1]=b; (m_mem)[i+2]=c; (m_mem)[i+3]=d;
80  (m_mem)[i+4]=e; (m_mem)[i+5]=f; (m_mem)[i+6]=g; (m_mem)[i+7]=h;
81  }
82  isaac();
83  m_cnt = 255;
84  }
85  void discard(unsigned long long z) {
86  for (; z != 0ULL; --z)
87  (*this).operator ()();
88  }
89  result_type operator()() {
90  return(!m_cnt-- ? (isaac(), m_cnt=(255), m_rsl[m_cnt]) :
91  m_rsl[m_cnt]);

93  static constexpr result_type min() { return 0; }
94  static constexpr result_type max() { return (size == 64) ? UINT64_MAX : UINT32_MAX; }
95  private:
96  void shuffle(result_type& a, result_type& b, result_type& c, result_type& d,
97  result_type& e, result_type& f, result_type& g, result_type& h){
98  if(size == 32){
99  a^=b<<11; d+=a; b+=c;
100  b^=c>>2; e+=b; c+=d;
101  c^=d<<8; f+=c; d+=e;
102  d^=e>>16; g+=d; e+=f;
103  e^=f<<10; h+=e; f+=g;
104  f^=g>>4; a+=f; g+=h;
105  g^=h<<8; b+=g; h+=a;
106  h^=a>>9; c+=h; a+=b;
107  } else if (size == 64) {
108  a-=e; f^=h>>9; h+=a;
109  b-=f; g^=a<<9; a+=b;
110  c-=g; h^=b>>23; b+=c;
111  d-=h; a^=c<<15; c+=d;
112  e-=a; b^=d>>14; d+=e;
113  f-=b; c^=e<<20; e+=f;
114  g-=c; d^=f>>17; f+=g;
115  h-=d; e^=g<<14; g+=h;
116  }
117  }
118  result_type ind(result_type* mm, result_type x) {
119  if(size == 32)
120  return (*(result_type*)((unsigned char*)(mm) + ((x) & ((255)<<2))));
121  else if(size == 64)
122  return (*(result_type*)((unsigned char*)(mm) + ((x) & ((255)<<3))));
123  }
124  void rngstep(result_type mix, result_type& a, result_type& b,
125  result_type*& mm, result_type*& m, result_type*& m2,
126  result_type*& r, result_type& x, result_type& y) {
127  x = *m;
128  a = (a^(mix)) + *(m2++);
129  *(m++) = y = ind(mm,x) + a + b;
130  *(r++) = b = ind(mm,y>>8) + x;
131  }
132  void isaac() {
133  result_type x,y;
134  result_type* mm = m_mem;
135  result_type* r = m_rsl;
136  result_type a = (m_a);
137  result_type b = (m_b + (++m_c));
138 
139  result_type* m = mm;
140  result_type* m2 = (m+(128));
141  result_type* mend = m2;
142 
143  for(; m<mend; ){
144  if(size == 32) {
145  rngstep((a<<13), a, b, mm, m, m2, r, x, y);
146  rngstep((a>>6) , a, b, mm, m, m2, r, x, y);
147  rngstep((a<<2) , a, b, mm, m, m2, r, x, y);
148  rngstep((a>>16), a, b, mm, m, m2, r, x, y);
149  } else {
150  rngstep(~(a^(a<<21)), a, b, mm, m, m2, r, x, y);
151  rngstep( a^(a>>5) , a, b, mm, m, m2, r, x, y);
152  rngstep( a^(a<<12) , a, b, mm, m, m2, r, x, y);
153  rngstep( a^(a>>33) , a, b, mm, m, m2, r, x, y);
154  }
155  }
156  m2 = mm;
157  for(; m2<mend; ) {
158  if(size == 32) {
159  rngstep((a<<13), a, b, mm, m, m2, r, x, y);
160  rngstep((a>>6) , a, b, mm, m, m2, r, x, y);
161  rngstep((a<<2) , a, b, mm, m, m2, r, x, y);
162  rngstep((a>>16), a, b, mm, m, m2, r, x, y);
163  } else {
164  rngstep(~(a^(a<<21)), a, b, mm, m, m2, r, x, y);
165  rngstep( a^(a>>5) , a, b, mm, m, m2, r, x, y);
166  rngstep( a^(a<<12) , a, b, mm, m, m2, r, x, y);
167  rngstep( a^(a>>33) , a, b, mm, m, m2, r, x, y);
168  }
169  }
170  m_b = b;
171  m_a = a;
172  }
173  private:
174  TIntType m_cnt;
175  TIntType m_rsl[256];
176  TIntType m_mem[256];
177  TIntType m_a;
178  TIntType m_b;
179  TIntType m_c;
180  };
181 }
