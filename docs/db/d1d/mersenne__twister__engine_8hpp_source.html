37  template <typename UIntType, size_t w, size_t n, size_t m, size_t r,

39  UIntType b, size_t t,
40  UIntType c, size_t l, UIntType f>
41  class mersenne_twister_engine
42  {
43  public:
44  using result_type = UIntType;
45  static constexpr size_t word_size = w;
46  static constexpr size_t state_size = n;
47  static constexpr size_t shift_size = m;
48  static constexpr size_t mask_bits = r;
49  static constexpr result_type xor_mask = a;
50  static constexpr size_t tempering_u = u;
51  static constexpr result_type tempering_d = d;
52  static constexpr size_t tempering_s = s;
53  static constexpr result_type tempering_b = b;
54  static constexpr size_t tempering_t = t;
55  static constexpr result_type tempering_c = c;
56  static constexpr size_t tempering_l = l;
57  static constexpr result_type initialization_multiplier = f;
58  static constexpr size_t MASK_LOWER = (1ull << mask_bits) - 1;
59  static constexpr size_t MASK_UPPER = (1ull << mask_bits);
60  static constexpr result_type default_seed = 5489u;
61 
62  explicit mersenne_twister_engine(result_type sd = default_seed) {
63  seed(sd);
64  }
65  void seed(result_type sd = default_seed) {
66  m_state[0] = seed;
67  m_index = state_size;
68  for (uint32_t i = 1; i < state_size; i++ ) {
69  m_state[i] = (initialization_multiplier * (m_state[i - 1] ^ (m_state[i - 1] >> 30)) + i);
70  }
71  }
72  result_type operator()() {
73  uint32_t y;
74  int i = m_index;
75  if ( m_index >= state_size ) {
76  twist();
77  i = m_index;
78  }
79  y = m_state[i];
80  m_index = i + 1;
81 
82  y ^= (m_state[i] >> tempering_u);
83  y ^= (y << tempering_s) & tempering_b;
84  y ^= (y << tempering_t) & tempering_c;
85  y ^= (y >> tempering_l);
86 
87  m_y = y;
88  return y;
89  }
90  void discard(unsigned long long z) {
91  for (; z != 0ULL; --z)
92  (*this).operator ()();
93  }
94  result_type last() {
95  return m_y;
96  }
97  static constexpr result_type min() { return 0; }
98  static constexpr result_type max() { return 2 ^ word_size - 1; }
99  private:
100  void twist() {
101  result_type i, x, xA;
102  for ( i = 0; i < state_size; i++ ){
103  x = (m_state[i] & MASK_UPPER) + (m_state[(i + 1) % state_size] & MASK_LOWER);
104  xA = x >> 1;
105  if ( x & 0x1 )
106  xA ^= xor_mask;
107 
108  m_state[i] = m_state[(i + shift_size) % state_size] ^ xA;
109  }
110  m_index = 0;
111  }
112  private:
113  result_type m_state[state_size+1];
114  uint16_t m_index;
115  result_type m_y;
116  };
117 }
