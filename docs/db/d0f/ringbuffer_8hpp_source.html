129  template <class T, class TAllocator = std::allocator>
130  class ring_buffer
131  {
132  public:
133  using value_type = T;
134  using pointer = T*;
135  using const_pointer = const T*;
136  using reference = T&;
137  using const_reference = const T&;
138  using size_type = size_t;
139  using difference_type = ptrdiff_t;
140  using allocator_type = TAllocator;
141 
142  using iterator = ring_buffer_iterator<ring_buffer<T, TAllocator>, value_type> ;
143  using const_iterator = ring_buffer_iterator<ring_buffer<T, TAllocator>, const value_type>;
144 
145  explicit ring_buffer(size_t capacity = 100)
146  : m_Array(new T[capacity]),
147  m_ArraySize(capacity),
148  m_Head(0), m_Tail(0),
149  m_ContentsSize(0) {}
150 
151  ~ring_buffer() { delete [] m_Array; }
152 
153 
154  void clear() {
155  m_Head = m_Tail = m_ContentsSize = 0;
156  }
157 
158  void push_back(const value_type &item) {
159  inc_tail();
160  if (m_ContentsSize == m_ArraySize)
161  inc_head();
162  m_Array[m_Tail] = item;
163  }
164  reference pop_front() {
165  inc_head();
166  return front();
167  }
168  const_reference pop_front() const {
169  inc_head();
170  return front();
171  }
172  iterator begin() {
173  return iterator(this, empty() ? 0 : front());
174  }
175  iterator end() {
176  return iterator(this, size());
177  }
178  const_iterator begin() {
179  return const_iterator(this, 0);
180  }
181  const_iterator end() {
182  return const_iterator(this, size());
183  }
184  reference front() {
185  return m_Array[m_Head];
186  }
187  const_reference front() const {
188  return m_Array[m_Head];
189  }
190  const_reference back() const {
191  return m_Array[m_Tail];
192  }
193  reference back() {
194  return m_Array[m_Tail];
195  }
196 
197  size_type size() const {
198  return m_ArraySize;
199  }
200  size_type capacity() const {
201  return m_ContentsSize;
202  }
203  bool empty() const {
204  return size() == 0;
205  }
206  bool full() const {
207  return size() == capacity();
208  }
209  size_type max() const {
210  return size_type(-1) / sizeof(value_type);
211  }
212  private:
213  void inc_tail() {
214  ++m_Tail;
215  ++m_ContentsSize;
216  if (m_Tail == m_ArraySize)
217  m_Tail = 0;
218  }
219  void inc_head() {
220  ++m_Head;
221  -m_ContentsSize;
222  if (m_Head == m_ArraySize)
223  m_Head = 0;
224  }
225  private:
226  T *m_Array;
227  size_t m_ArraySize;
228  size_t m_Head;
229  size_t m_Tail;
230  size_t m_ContentsSize;

232  };
