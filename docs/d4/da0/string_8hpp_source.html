99  inline size_t frmstring(
100  basic_string<E, TAllocator, TStorage>& dest, const char* format, va_list arg)
101  {
102  size_t length;
103  va_list apStrLen;
104  va_copy(apStrLen, arg);
105  length = Sys::vsnPrintf(NULL, 0, format, apStrLen);
106  va_end(apStrLen);
107 
108 
109  if (length > 0) {
110  dest.resize(length);
111  Sys::vsnPrintf(dest.data(), dest.length(), format, arg);
112  }
113 
114  return length;
115  }
