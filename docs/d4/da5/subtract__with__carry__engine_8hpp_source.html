39  template<typename Tp, Tp m, Tp a, Tp c, bool s>
40  struct _mod<Tp, m, a, c, true, s> {
41  static Tp calc(Tp x) {
42  Tp res = a * x + c;
43  if (m) res %= m;
44  return res;
45  }
46  };
47  template<typename Tp, Tp m, Tp a = 1, Tp c = 0>
48  inline Tp mod(Tp x) { return mod<Tp, m, a, c>::calc(x); }
49  }
50  template<class UIntType, size_t word_size, size_t short_lag, size_t long_lag>
51  class subtract_with_carry_engine
52  {
53  public:
54  using result_type = UIntType;
55  static constexpr size_t default_seed = 19780503u;
56 
57  subtract_with_carry_engine(size_t sd = default_seed) {
58  seed(sd);
59  }

61  void seed(size_t sd = default_seed) {
62  linear_congruential_engine<result_type,
63  40014u, 0u, 2147483563u> e(sd == 0u ? default_seed : sd);
64 
65  result_type mds = (UIntType(1) << word_size);
66  for(size_t i = 0; i < long_lag; ++i)
67  m_x[i] = internal::mod<result_type, 1, 0, mds>(e());
68  m_carry = (m_x[long_lag-1] == 0);
69  m_p = 0;
70  }
71  void discard(unsigned long long z) {
72  for (; z != 0ULL; --z)
73  (*this).operator ()();
74  }
75  result_type last() {
76  return m_xlast;
77  }
78  result_type operator()() {
79  int short_index = m_p - short_lag;
80  short_index += (short_index < 0) ? long_lag : 0;
81  if (m_x[short_index] >= m_x[m_p] + m_carry) {
82  m_xlast = m_x[short_index] - m_x[m_p] - m_carry;
83  m_carry = 0;
84  } else {
85  m_xlast = modulus - m_x[m_p] - m_carry + m_x[short_index];
86  m_carry = 1;
87  }
88  m_x[m_p] = m_xlast; ++m_p;
89  if(m_p >= long_lag)
90  m_p = 0;
91  return m_xlast;
92  }
93  static constexpr result_type min() { return 0; }
94  static constexpr result_type max() { return (UIntType(1) << word_size) - 1; }
95  private:
96  result_type m_carry;
97  result_type m_x[long_lag];
98  size_t m_p;
99  result_type m_xlast;
100  };
