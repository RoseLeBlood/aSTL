143  {

181  iterator insert(iterator pos, value_type* v)
182  {
183  return insert_after(previous(pos), v);
184  }
185  iterator insert_after(iterator pos, value_type* v)
186  {
187  link_after(v, pos.node());
188  return iterator(v);
189  }
190  // @note use erase_after if possible, it's quicker.
191  iterator erase(iterator pos)
192  {
193  pos = previous(pos);
194  erase_after(pos);
195  return ++pos;
196  }
197  void erase_after(iterator pos)
198  {
199  unlink_after(pos.node());
200  }

202  // @return iterator prev such that ++prev == nextIt
203  static iterator previous(iterator nextIt)
204  {
205  assert(nextIt.node()->in_list());
206  iterator prevIt = nextIt;
207  while (nextIt.node() != prevIt.next())
208  ++prevIt;
209  return prevIt;
210  }
211  static const_iterator previous(const_iterator nextIt)
212  {
213  assert(nextIt.node()->in_list());
214  const_iterator prevIt = nextIt;
215  while (nextIt.node() != prevIt.next())
216  ++prevIt;
217  return prevIt;
218  }
219  // O(1)
220  static iterator get_iterator(value_type* v)
221  {
222  assert(v->in_list());
223  return iterator(v);
224  }
225  // O(1)
226  static const_iterator get_iterator(const value_type* v)
227  {
228  assert(v->in_list());
229  return const_iterator(v);
230  }

232  iterator erase(iterator first, iterator last)
233  {
234  if (first != last)
235  {
236  while (first.next() != last.node())
237  erase_after(first);
238  erase_after(previous(first));
239  }
240  return first;
241  }
242 
243  void clear()
244  {
245  erase(begin(), end());
246  }
247 
248  static void remove(value_type* v)
249  {

251  }

257  }
