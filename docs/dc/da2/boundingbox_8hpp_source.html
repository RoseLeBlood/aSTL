68  contains_t contains(const boundingbox<T>& box) {
69  if ((m_vMax.x < box.m_vMin.x || m_vMin.x > box.m_vMax.x) ||
70  (m_vMax.y < box.m_vMin.y || m_vMin.y > box.m_vMax.y) ||
71  (m_vMax.z < box.m_vMin.z || m_vMin.z > box.m_vMax.z)) {
72  return contains_t::Disjoint;
73  }
74  if ((m_vMin.x <= box.m_vMin.x && m_vMax.x >= box.m_vMax.x) &&
75  (m_vMin.y <= box.m_vMin.y && m_vMax.y >= box.m_vMax.y) &&
76  (m_vMin.z <= box.m_vMin.z && m_vMax.z >= box.m_vMax.z)) {
77  return contains_t::Contains;
78  }
79  return contains_t::Intersects;
80  }
81  contains_t contains (vector3<T> point)
82  {
83  if ((m_vMin.x <= point.x && m_vMax.x >= point.x) &&
84  (m_vMin.y <= point.y && m_vMax.y >= point.y) &&
85  (m_vMin.z <= point.z && m_vMax.z >= point.z)) {
86  return contains_t::Contains;
87  }
88  return contains_t::Disjoint;
89  }

94  corners[2] = vector3<T> (m_vMax.x, m_vMin.y, m_vMax.z);
95  corners[3] = vector3<T> (m_vMin.x, m_vMin.y, m_vMax.z);
