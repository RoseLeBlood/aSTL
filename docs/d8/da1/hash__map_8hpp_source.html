257  {

411  node* find_for_insert(const key_type& key, hash_value_t* out_hash)
412  {
413  if (m_capacity == 0)
414  return 0;
415 
416  const hash_value_t hash = hash_func(key);
417  *out_hash = hash;
418  uint32_t i = hash & m_capacityMask;
419 
420  node* n = m_nodes + i;
421  if (n->hash == hash && m_keyEqualFunc(key, n->data.first))
422  return n;
423 
424  node* freeNode(0);
425  if (n->is_deleted())
426  freeNode = n;
427  uint32_t numProbes(0);
428  // Guarantees loop termination.
429  assert(m_numUsed < m_capacity);
430  while (!n->is_unused())
431  {
432  ++numProbes;
433  i = (i + numProbes) & m_capacityMask;
434  n = m_nodes + i;
435  if (compare_key(n, key, hash))
436  return n;
437  if (n->is_deleted() && freeNode == 0)
438  freeNode = n;
439  }
440  return freeNode ? freeNode : n;
441  }
442  node* lookup(const key_type& key) const
443  {
444  const hash_value_t hash = hash_func(key);
445  uint32_t i = hash & m_capacityMask;
446  node* n = m_nodes + i;
447  if (n->hash == hash && m_keyEqualFunc(key, n->data.first))
448  return n;
449 
450  uint32_t numProbes(0);
451  // Guarantees loop termination.
452  assert(m_capacity == 0 || m_numUsed < m_capacity);
453  while (!n->is_unused())
454  {
455  ++numProbes;
456  i = (i + numProbes) & m_capacityMask;
457  n = m_nodes + i;

461  }

465  static void rehash(int new_capacity, node* new_nodes, int capacity, const node* nodes, bool destruct_original)
466  {
467  const node* it = nodes;
468  const node* itEnd = nodes + capacity;
469  const uint32_t mask = new_capacity - 1;
470  while (it != itEnd)
471  {
472  if (it->is_occupied())
473  {
474  const hash_value_t hash = it->hash;
475  uint32_t i = hash & mask;
476 
477  node* n = new_nodes + i;
478  uint32_t numProbes(0);
479  while (!n->is_unused())
480  {
481  ++numProbes;
482  i = (i + numProbes) & mask;
483  n = new_nodes + i;
484  }
485  std::copy_construct(&n->data, it->data);
486  n->hash = hash;
487  if (destruct_original)
488  std::destruct(&it->data);
489  }
490  ++it;
491  }
492  }
