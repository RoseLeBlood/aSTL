59  class alignas(16) vecf_sse : public object {
60  public:
61  vecf_sse () { m_xmm = _mm_set1_ps(0); }
62  vecf_sse (__m128 v) : m_xmm (v) {}
63  vecf_sse (const vecf_sse &v) { m_xmm = v.m_xmm; }
64 
65  vecf_sse (float v) { m_xmm = _mm_set1_ps(v); }
66  vecf_sse (float x, float y) { m_xmm = _mm_set_ps(0,0,y,x); }
67  vecf_sse (float x, float y, float z) { m_xmm = _mm_set_ps(0,z,y,x); }
68  vecf_sse (float x, float y, float z, float w) { m_xmm = _mm_set_ps(w,z,y,x); }
69 
70  vecf_sse (const vec2f &v) { m_xmm = _mm_set_ps(0,0,v.y,v.x); }
71  vecf_sse (const vec3f &v) { m_xmm = _mm_set_ps(0,v.z,v.y,v.x); }
72  vecf_sse (const vec4f &v) { m_xmm = _mm_set_ps(v.w,v.z,v.y,v.x); }
73 
74  vecf_sse (const float *v) { m_xmm = _mm_load_ps(v); }
75 
76 
77  float X () const { return ((float*)&m_xmm)[0]; }
78  float Y () const { return ((float*)&m_xmm)[1]; }
79  float Z () const { return ((float*)&m_xmm)[2]; }
80  float W () const { return ((float*)&m_xmm)[3]; }
81 
82  __m128 M() { return m_xmm; }
83 
84  void operator *= (const vecf_sse &v) { m_xmm = _mm_mul_ps(m_xmm, v.m_xmm); }
85  void operator += (const vecf_sse &v) { m_xmm = _mm_add_ps(m_xmm, v.m_xmm); }
86  void operator -= (const vecf_sse &v) { m_xmm = _mm_sub_ps(m_xmm, v.m_xmm); }
87  void operator /= (const vecf_sse &v) { m_xmm = _mm_div_ps(m_xmm, v.m_xmm); }
88 
89  operator vec2f () { return vec2f(X(), Y()); }
90  operator vec3f () { return vec3f(X(), Y(), Z()); }
91  operator vec4f () { return vec4f(X(), Y(), Z(), W()); }
92 
93  private:
94  __m128 m_xmm;
95  };
