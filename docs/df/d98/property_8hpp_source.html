40  template <typename T, class C>
41  class iproperty_base
42  {
43  public:
44  using value_type = T;
45  using cont_type = C;
46  using self_type = iproperty_base<value_type, cont_type>;
47 
48  iproperty_base(cont_type* ctype, const T(cont_type::*pget)())
49  : m_pContainer(ctype) {
50  m_pSetter = (void*)0;
51  m_pGetter = pget;
52  }
53  iproperty_base(cont_type* ctype, void (cont_type::*pset)(const T& value))
54  : m_pContainer(ctype) {
55  m_pSetter = pset;
56  m_pGetter = (void*)0;
57  }
58  iproperty_base(cont_type* ctype, void (cont_type::*pset)(const T& value), const T(cont_type::*pget)())
59  : m_pContainer(ctype) {
60  m_pSetter = pset;
61  m_pGetter = pget;
62  }
63  value_type operator = (const value_type& value) {
64  assert(m_pContainer != NULL);
65  assert(m_pSetter != NULL);
66  (m_pContainer->*m_pSetter)(value);
67  return value;
68  }
69  explicit operator value_type() {
70  assert(m_pContainer != NULL);
71  assert(m_pGetter != NULL);
72  return (m_pContainer->*m_pGetter)();
73  }
74  private:
75  cont_type* m_pContainer;
76  void (cont_type::*m_pSetter)(const value_type& value);
77  const value_type (cont_type::*m_pGetter)();
78  };

82  template <typename T, bool read = true, bool write = false>
83  class value_property
84  {
85  public:
86  using value_type = T;
87  using self_type = value_property<T, read, write>;
88 
89  value_property() { }
90  explicit value_property(value_type value)
91  : m_tValue(value) {
92  }
93  bool can_read() { return read; }
94  bool can_write() { return write; }
95 
96  value_type operator = (const value_type& value) {
97  assert(write);
98  set(value);
99  return value;
100  }
101  explicit operator value_type() {
102  assert(read);
103  return get();
104  }
105  protected:
106  virtual void set(const value_type& value) { m_tValue = value; }
107  virtual const value_type get() { return m_tValue; }
108  private:
109  value_type m_tValue;
110  };
