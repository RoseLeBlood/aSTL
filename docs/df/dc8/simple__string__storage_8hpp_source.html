40  template<typename E, class TAllocator> class simple_string_storage
41  {
42  public:
43  using value_type = E;
44  using size_type = int;
45  using allocator_type = TAllocator;
46  using const_iterator = const value_type* ;
47  static const unsigned long kGranularity = 32;
48 
49  explicit simple_string_storage(const allocator_type& allocator)
50  : m_length(0), m_allocator(allocator)
51  {
52  m_data = construct_string(0, m_capacity);
53  }
54  simple_string_storage(const value_type* str, const allocator_type& allocator)
55  : m_allocator(allocator)
56  {
57  const int len = strlen(str);
58  m_data = construct_string(len, m_capacity);
59  std::Sys::MemCpy(m_data, str, len*sizeof(value_type));
60  m_length = len;
61  m_data[len] = 0;
62  }
63  simple_string_storage(const value_type* str, size_type len, const allocator_type& allocator)
64  : m_allocator(allocator)
65  {
66  m_data = construct_string(len, m_capacity);
67  std::Sys::MemCpy(m_data, str, len*sizeof(value_type));
68  m_length = len;
69  m_data[len] = 0;
70  }
71  simple_string_storage(size_type len, const allocator_type& allocator)
72  : m_allocator(allocator)
73  {
74  m_data = construct_string(len, m_capacity);
75  m_length = len;
76  m_data[len] = 0;
77  }
78  simple_string_storage(const simple_string_storage& rhs, const allocator_type& allocator)
79  : m_data(0), m_capacity(0), m_length(0), m_allocator(allocator)

82  }

97  void assign(const value_type* str, size_type len)
98  {
99  //assert(str != m_data);
100  if (m_capacity <= len + 1)
101  {
102  release_string();
103  m_data = construct_string(len, m_capacity);
104  }
105  std::Sys::MemCpy(m_data, str, len*sizeof(value_type));
106  m_length = len;
107  m_data[len] = 0;
108  }
109  void resize(size_type len)
110  {
111  const size_type prevLen = length();
112  const size_type newLen = len;
113  if (m_capacity <= newLen + 1)
114  {
115  size_type newCapacity;
116  value_type* newData = construct_string(newLen, newCapacity);
117  std::Sys::MemCpy(newData, m_data, prevLen * sizeof(value_type));
118  release_string();
119  m_data = newData;
120  m_capacity = newCapacity;
121  }
122  m_data[newLen] = 0;
123  m_length = newLen;
124  assert(invariant());
125  }
126 
127  void append(const value_type* str, size_type len)
128  {
129  const size_type prevLen = length();
130  const size_type newLen = prevLen + len;
131  if (m_capacity <= newLen + 1)
132  {
133  size_type newCapacity;
134  value_type* newData = construct_string(newLen, newCapacity);
135  Sys::MemCpy(newData, m_data, prevLen * sizeof(value_type));
136  release_string();
137  m_data = newData;
138  m_capacity = newCapacity;
139  }
140  Sys::MemCpy(m_data + prevLen, str, len * sizeof(value_type));
141  m_data[newLen] = 0;
142  m_length = newLen;
143  assert(invariant());
144  }

149  }

151  inline size_type length() const

153  return m_length;

156  inline size_type capacity() const { return m_capacity; }
157 
158  void clear()
159  {
160  release_string();
161  m_data = construct_string(0, m_capacity);
162  m_length = 0;
163  }
164 
165  const allocator_type& get_allocator() const { return m_allocator; }
166 
167  void make_unique(size_type) {}
168  value_type* get_data() { return m_data; }
169 
170  protected:
171  bool invariant() const
172  {
173  assert(m_data);
174  assert(m_length <= m_capacity);
175  if (length() != 0)
176  assert(m_data[length()] == 0);
177  return true;
178  }
179  private:
180  value_type* construct_string(size_type capacity, size_type& out_capacity)
181  {
182  value_type* data(0);
183  if (capacity != 0)
184  {
185  capacity = (capacity+kGranularity-1) & ~(kGranularity-1);
186  if (capacity < kGranularity)
187  capacity = kGranularity;
188 
189  const size_type toAlloc = sizeof(value_type)*(capacity + 1);
190  void* mem = m_allocator.allocate(toAlloc);
191  data = static_cast<value_type*>(mem);
192  }
193  else // empty string, no allocation needed. Use our internal buffer.
194  {
195  data = &m_end_of_data;
196  }
197 
198  out_capacity = capacity;
199  *data = 0;
200  return data;
201  }
202  void release_string()
203  {
204  if (m_capacity != 0)
205  {
206  assert(m_data != &m_end_of_data);
207  m_allocator.deallocate(m_data, m_capacity);
208  }
209  }
210  private:
211  E* m_data;
212  E m_end_of_data;
213  size_type m_capacity;
214  size_type m_length;
215  TAllocator m_allocator;
216  };
