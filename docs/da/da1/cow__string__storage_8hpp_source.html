38  struct string_rep
39  {
40  void add_ref() {
41  SPINLOCK(loc) {
42  ++refs;
43  }
44  }
45  bool release() {
46  SPINLOCK(loc) {
47  --refs;
48  }
49  return refs <= 0;
50  }
51  void init(short new_capacity = 0) {
52  refs = 1;
53  size = 0;
54  capacity = new_capacity;
55  }
56 
57  short refs;
58  short size;
59  short capacity;
60  std::spinlock loc;
61 
62  static const size_t kMaxCapacity = (1 << (sizeof(short) << 3)) >> 1;
63  };

110  cow_string_storage(const cow_string_storage& rhs, const allocator_type& allocator)
111  : m_data(rhs.m_data), m_allocator(allocator)
112  {
113  if (rhs.is_dynamic())
114  {
115  get_rep()->add_ref();
116  }
117  else
118  {
119  const int len = rhs.length();
120  construct_string(len);
121  Sys::MemCpy(m_data, rhs.c_str(), len*sizeof(value_type));
122  assert(len < string_rep::kMaxCapacity);
123  get_rep()->size = static_cast<short>(len);
124  m_data[len] = 0;
125  }
126  }

128  {
129  if (!is_dynamic())
130  RDE_ASSERT(get_rep()->refs == 1);
131  release_string();
132  }

134  {

144  void assign(const value_type* str, size_type len)
145  {
146  assert(str != m_data);
147  release_string();

149  Sys::MemCpy(m_data, str, len*sizeof(value_type));
150  get_rep()->size = short(len);

153  void resize( size_type len)
154  {
155  const size_type prevLen = length();
156  const size_type newCapacity = len;
157  make_unique(newCapacity);
158  string_rep* rep = get_rep();
159  assert(rep->capacity >= short(newCapacity));
160  const size_type newLen = prevLen + len;
161  assert(short(newLen) <= rep->capacity);
162  m_data[newLen] = 0;
163  rep->size = short(newLen);
164  }
165  void append(const value_type* str, size_type len)
166  {
167  const size_type prevLen = length();
168  const size_type newCapacity = prevLen + len;
169  make_unique(newCapacity);
170  string_rep* rep = get_rep();
171  assert(rep->capacity >= short(newCapacity));
172  const size_type newLen = prevLen + len;
173  assert(short(newLen) <= rep->capacity);
174  Sys::MemCpy(m_data + prevLen, str, len * sizeof(value_type));
175  m_data[newLen] = 0;
176  rep->size = short(newLen);
177  }

206  protected:
207  bool invariant() const
208  {
209  assert(m_data);
210  string_rep* rep = get_rep();
211  assert(rep->refs >= 1);
212  assert(rep->size <= rep->capacity);
213  if (length() != 0)
214  assert(m_data[length()] == 0);
215  return true;
216  }
217  void make_unique(size_type capacity_hint)
218  {
219  string_rep* rep = get_rep();
220  assert(rep->refs >= 1);
221 
222  if (capacity_hint != 0)
223  {
224  ++capacity_hint;
225  capacity_hint = (capacity_hint+kGranularity-1) & ~(kGranularity-1);
226  if (capacity_hint < kGranularity)
227  capacity_hint = kGranularity;
228  }
229  assert(capacity_hint < string_rep::kMaxCapacity);
230  // Reallocate string only if we truly need to make it unique
231  // (it's shared) or if our current buffer is too small.
232  if (rep->refs > 1 || short(capacity_hint) > rep->capacity)
233  {
234  if (capacity_hint > 0)
235  {
236  const size_type toAlloc = sizeof(string_rep) + sizeof(value_type)*capacity_hint;
237  void* newMem = m_allocator.allocate(toAlloc);
238  string_rep* newRep = reinterpret_cast<string_rep*>(newMem);
239  newRep->init(short(capacity_hint));
240  value_type* newData = reinterpret_cast<value_type*>(newRep + 1);
241  Sys::MemCpy(newData, m_data, rep->size*sizeof(value_type));
242  newRep->size = rep->size;
243  newData[rep->size] = 0;
244  release_string();
245  m_data = newData;
246  }
247  else
248  {
249  release_string();
250  string_rep* rep = reinterpret_cast<string_rep*>(m_buffer);
251  rep->init();
252  m_data = reinterpret_cast<value_type*>(rep + 1);
253  *m_data = 0;
254  }
255  }
256  }
257  inline E* get_data() { return m_data; }

259  private:
260  inline string_rep* get_rep() const
261  {
262  return reinterpret_cast<string_rep*>(m_data) - 1;
263  }
264  void construct_string(size_type capacity)
265  {
266  if (capacity != 0)
267  {
268  ++capacity;
269  capacity = (capacity+kGranularity-1) & ~(kGranularity-1);
270  if (capacity < kGranularity)
271  capacity = kGranularity;
272  assert(capacity < string_rep::kMaxCapacity);
273 
274  const size_type toAlloc = sizeof(string_rep) + sizeof(value_type)*capacity;
275  void* mem = m_allocator.allocate(toAlloc);
276  string_rep* rep = reinterpret_cast<string_rep*>(mem);
277  rep->init(static_cast<short>(capacity));
278  m_data = reinterpret_cast<value_type*>(rep + 1);
279  }
280  else // empty string, no allocation needed. Use our internal buffer.
281  {
282  string_rep* rep = reinterpret_cast<string_rep*>(m_buffer);
283  rep->init();
284  m_data = reinterpret_cast<value_type*>(rep + 1);
285  }
286  *m_data = 0;
287  }
288  void release_string()
289  {
290  string_rep* rep = get_rep();
291  if (rep->release() && rep->capacity != 0)
292  {
293  assert(is_dynamic());
294  m_allocator.deallocate(rep, rep->capacity);
295  }
296  }
297  bool is_dynamic() const
298  {
299  return get_rep() != reinterpret_cast<const string_rep*>(&m_buffer[0]);
300  }
301 
302  E* m_data;
303  char m_buffer[sizeof(string_rep)+2];
304  TAllocator m_allocator;
305  };
306 }
