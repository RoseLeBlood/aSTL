41  template <typename T>
42  class scoped_ptr {
43  public:
44  using value_type = T;
45  using pointer = value_type*;
46  using const_value_type = const value_type;
47  using self_type = scoped_ptr<value_type>;
48 
49  explicit scoped_ptr( pointer pValue = 0 )
50  : m_pPointer( pValue ) { }
51  explicit scoped_ptr( const auto_ptr<value_type>& pPtr )
52  : m_pPointer( pPtr.release() ) { }
53  ~scoped_ptr() { }
54 
55  void swap(self_type& b) {
56  std::swap<value_type*>(m_pPointer, b.m_pPointer);
57  }
58  void reset( pointer pValue = 0)
59  { self_type(pValue).swap(*this); }
60  pointer get()
61  { return m_pPointer; }
62 
63  const pointer operator -> () {
64  assert(m_pPointer != 0);
65  return m_pPointer;
66  }
67  const_value_type& operator *() {
68  assert(m_pPointer != 0);
69  return *m_pPointer;
70  }
71  operator bool() {
72  return m_pPointer != 0;
73  }
74  private:
75  pointer m_pPointer;
76  };
77  template<class T>
78  inline void swap(scoped_ptr<T> & a, scoped_ptr<T> & b) {
